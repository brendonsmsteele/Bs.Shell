using UnityEditor;
using UnityEngine;

namespace Bs.Shell.UI
{
    [CustomEditor(typeof(TEMPLATEDataEvent))]
    public class TEMPLATEDataEventEditor : Editor
    {
        public override void OnInspectorGUI()
        {
            DrawDefaultInspector();
            TEMPLATEDataEvent myTarget = (TEMPLATEDataEvent)target;

            if (Application.isPlaying && myTarget.fakeData != null)
            {
                if (GUILayout.Button("Raise"))
                    myTarget.Raise(myTarget.fakeData);
            }
        }
    }
}
